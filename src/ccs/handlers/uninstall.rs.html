1//! Uninstall handler
2//!
3//! Removes all ccs artifacts: hooks, wrapper, config directory, sync repo, and binary.
4
5use anyhow::{Context, Result};
6use colored::Colorize;
7use inquire::Confirm;
8
9use crate::config::ConfigManager;
10use crate::sync::MultiRepoState;
11
12/// Run the uninstall flow
13pub fn handle_uninstall(force: bool) -> Result<()> {
14    println!();
15    println!("{}", "🗑️  卸载 Claude Code Sync".red().bold());
16    println!("{}", "━━━━━━━━━━━━━━━━━━━━━━━━━".red());
17    println!();
18
19    // Gather info about what exists
20    let config_dir = ConfigManager::config_dir().ok();
21    let sync_repo_path = MultiRepoState::load()
22        .ok()
23        .and_then(|s| s.active().map(|r| r.sync_repo_path.clone()));
24    let current_exe = std::env::current_exe().ok();
25
26    // Show what will be removed
27    println!("{}", "将要执行以下操作:".cyan());
28    println!("   1. 卸载 Claude Code hooks");
29    println!("   2. 删除 wrapper 脚本 (claude-sync)");
30    if let Some(ref dir) = config_dir {
31        println!("   3. 删除配置目录 ({})", dir.display());
32    }
33    if let Some(ref repo) = sync_repo_path {
34        println!("   4. 删除同步仓库 ({}) [需单独确认]", repo.display());
35    }
36    if let Some(ref exe) = current_exe {
37        println!("   5. 删除 ccs 二进制 ({}) [需单独确认]", exe.display());
38    }
39    println!();
40
41    if !force {
42        let confirm = Confirm::new("确认卸载?")
43            .with_default(false)
44            .prompt()
45            .unwrap_or(false);
46
47        if !confirm {
48            println!("{}", "已取消卸载。".yellow());
49            return Ok(());
50        }
51        println!();
52    }
53
54    // Step 1: Uninstall hooks
55    println!("{}", "1. 卸载 hooks...".cyan());
56    match crate::handlers::hooks::handle_hooks_uninstall() {
57        Ok(()) => {}
58        Err(e) => println!("   {} {}", "跳过:".yellow(), e),
59    }
60
61    // Step 2: Remove wrapper
62    println!("{}", "2. 删除 wrapper 脚本...".cyan());
63    match crate::handlers::wrapper::handle_wrapper_uninstall() {
64        Ok(()) => {}
65        Err(e) => println!("   {} {}", "跳过:".yellow(), e),
66    }
67
68    // Step 3: Remove config directory
69    if let Some(ref dir) = config_dir {
70        println!("{}", "3. 删除配置目录...".cyan());
71        if dir.exists() {
72            std::fs::remove_dir_all(dir)
73                .with_context(|| format!("删除配置目录失败: {}", dir.display()))?;
74            println!("   {} {}", "✓".green(), dir.display());
75        } else {
76            println!("   {} 配置目录不存在，跳过", "!".yellow());
77        }
78    }
79
80    // Step 4: Remove sync repo (requires separate confirmation)
81    if let Some(ref repo) = sync_repo_path {
82        if repo.exists() {
83            println!();
84            println!(
85                "{}",
86                "⚠️  同步仓库可能包含未推送的对话历史".yellow().bold()
87            );
88            println!("   路径: {}", repo.display());
89
90            let delete_repo = if force {
91                true
92            } else {
93                Confirm::new("是否删除同步仓库?")
94                    .with_default(false)
95                    .prompt()
96                    .unwrap_or(false)
97            };
98
99            if delete_repo {
100                std::fs::remove_dir_all(repo)
101                    .with_context(|| format!("删除同步仓库失败: {}", repo.display()))?;
102                println!("   {} 同步仓库已删除", "✓".green());
103            } else {
104                println!("   {} 保留同步仓库", "!".yellow());
105            }
106        }
107    }
108
109    // Step 5: Remove binary (requires separate confirmation)
110    if let Some(ref exe) = current_exe {
111        println!();
112        let delete_binary = if force {
113            true
114        } else {
115            Confirm::new(&format!("是否删除 ccs 二进制 ({})?", exe.display()))
116                .with_default(false)
117                .prompt()
118                .unwrap_or(false)
119        };
120
121        if delete_binary {
122            // Also remove wrapper in same directory (in case step 2 missed it)
123            let exe_dir = exe.parent();
124            if let Some(dir) = exe_dir {
125                for name in &["claude-sync", "claude-sync.bat", "claude-sync.ps1"] {
126                    let wrapper = dir.join(name);
127                    if wrapper.exists() {
128                        std::fs::remove_file(&wrapper).ok();
129                    }
130                }
131            }
132
133            std::fs::remove_file(exe)
134                .with_context(|| format!("删除二进制失败: {}", exe.display()))?;
135            println!("   {} ccs 二进制已删除", "✓".green());
136        } else {
137            println!("   {} 保留 ccs 二进制", "!".yellow());
138        }
139    }
140
141    println!();
142    println!("{}", "✓ 卸载完成".green().bold());
143    println!();
144
145    Ok(())
146}
