55/// Push configuration to sync repository (only copy files, no commit/push)
56/// Returns the list of synced files
57pub fn push_config_files(settings: &ConfigSyncSettings) -> Result<Vec<String>> {
58    let device_name = settings.get_device_name();
59    log::info!("Pushing configuration files for device: {}", device_name);
60
61    let sync_state = SyncState::load()?;
62    let sync_repo = sync_state.sync_repo_path.clone();
63    let claude = claude_dir()?;
64    let target_dir = device_config_dir(&sync_repo, &device_name);
65
66    // Create target directory
67    fs::create_dir_all(&target_dir)
68        .with_context(|| format!("Failed to create config dir: {}", target_dir.display()))?;
69
70    let mut synced_files = Vec::new();
71
72    // Sync settings.json (without hooks)
73    if settings.sync_settings {
74        let settings_path = claude.join("settings.json");
75        if settings_path.exists() {
76            let content = fs::read_to_string(&settings_path)?;
77
78            // Parse and remove hooks
79            if let Ok(mut json) = serde_json::from_str::<serde_json::Value>(&content) {
80                // Save full version with hooks
81                let full_path = target_dir.join("settings-full.json");
82                fs::write(&full_path, &content)?;
83                synced_files.push("settings-full.json".to_string());
84
85                // Remove hooks for portable version
86                if let Some(obj) = json.as_object_mut() {
87                    obj.remove("hooks");
88                }
89                let portable_content = serde_json::to_string_pretty(&json)?;
90                let portable_path = target_dir.join("settings.json");
91                fs::write(&portable_path, portable_content)?;
92                synced_files.push("settings.json".to_string());
93            } else {
94                // Just copy as-is if not valid JSON
95                fs::copy(&settings_path, target_dir.join("settings.json"))?;
96                synced_files.push("settings.json".to_string());
97            }
98        }
99    }
100
101    // Sync CLAUDE.md
102    if settings.sync_claude_md {
103        let claude_md_path = claude.join("CLAUDE.md");
104        if claude_md_path.exists() {
105            fs::copy(&claude_md_path, target_dir.join("CLAUDE.md"))?;
106            synced_files.push("CLAUDE.md".to_string());
107        }
108    }
109
110    // Sync hooks folder
111    if settings.sync_hooks {
112        let hooks_dir = claude.join("hooks");
113        if hooks_dir.exists() && hooks_dir.is_dir() {
114            let target_hooks = target_dir.join("hooks");
115            if target_hooks.exists() {
116                fs::remove_dir_all(&target_hooks)?;
117            }
118            copy_dir_recursive(&hooks_dir, &target_hooks)?;
119            synced_files.push("hooks/".to_string());
120        }
121    }
122
123    // Sync skills list
124    if settings.sync_skills_list {
125        let skills_dir = claude.join("skills");
126        if skills_dir.exists() && skills_dir.is_dir() {
127            let skills_list = generate_skills_list(&skills_dir)?;
128            let skills_json = serde_json::to_string_pretty(&skills_list)?;
129            fs::write(target_dir.join("installed_skills.json"), skills_json)?;
130            synced_files.push("installed_skills.json".to_string());
131        }
132
133        // Also sync plugins list if exists
134        let plugins_path = claude.join("plugins/installed_plugins.json");
135        if plugins_path.exists() {
136            fs::copy(&plugins_path, target_dir.join("installed_plugins.json"))?;
137            synced_files.push("installed_plugins.json".to_string());
138        }
139    }
140
141    // Save sync metadata
142    let sync_info = DeviceSyncInfo {
143        device: device_name.clone(),
144        platform: Platform::current().to_string(),
145        last_sync: chrono::Utc::now().to_rfc3339(),
146    };
147    let info_json = serde_json::to_string_pretty(&sync_info)?;
148    fs::write(target_dir.join(".sync-info.json"), info_json)?;
149
150    Ok(synced_files)
151}

153/// Push configuration to sync repository (with commit and push)
154pub fn handle_config_push(settings: &ConfigSyncSettings) -> Result<()> {
155    let device_name = settings.get_device_name();
156
157    let synced_files = push_config_files(settings)?;
158
159    // Commit and push
160    if !synced_files.is_empty() {
161        let sync_state = SyncState::load()?;
162        let sync_repo = sync_state.sync_repo_path.clone();
163        let message = format!("Sync config from {}", device_name);
164        let repo = scm::open(&sync_repo)?;
165
166        // Stage all changes
167        repo.stage_all()?;
168
169        // Check if there are changes to commit
170        if repo.has_changes()? {
171            repo.commit(&message)?;
172
173            // Push to remote if available
174            if sync_state.has_remote {
175                let branch = repo.current_branch()?;
176                repo.push("origin", &branch)?;
177            }
178
179            println!("{}", "✓ 配置已推送".green());
180            for file in &synced_files {
181                println!("  - {}", file);
182            }
183        } else {
184            println!("{}", "配置无变化".dimmed());
185        }
186    } else {
187        println!("{}", "⚠️  没有找到可同步的配置文件".yellow());
188    }
189
190    Ok(())
191}

193/// List available device configurations
194pub fn handle_config_list() -> Result<()> {
195    let sync_state = SyncState::load()?;
196    let configs = configs_dir(&sync_state.sync_repo_path);
197
198    if !configs.exists() {
199        println!("{}", "没有找到配置同步目录".yellow());
200        println!("运行 {} 推送当前设备配置", format!("{} config push", BINARY_NAME).cyan());
201        return Ok(());
202    }
203
204    let current_device = ConfigSyncSettings::default().get_device_name();
205
206    println!("{}", "可用的设备配置:".bold());
207    println!();
208
209    let mut found_any = false;
210    for entry in fs::read_dir(&configs)? {
211        let entry = entry?;
212        if !entry.file_type()?.is_dir() {
213            continue;
214        }
215
216        let device_name = entry.file_name().to_string_lossy().to_string();
217        found_any = true;
218
219        // Read sync info
220        let info_path = entry.path().join(".sync-info.json");
221        let sync_info: Option<DeviceSyncInfo> = if info_path.exists() {
222            fs::read_to_string(&info_path)
223                .ok()
224                .and_then(|s| serde_json::from_str(&s).ok())
225        } else {
226            None
227        };
228
229        // Display device
230        if device_name == current_device {
231            println!("  {} (当前设备)", device_name.green());
232        } else {
233            println!("  {}", device_name.cyan());
234        }
235
236        if let Some(info) = sync_info {
237            println!("    平台: {}", info.platform);
238            println!("    最后同步: {}", info.last_sync);
239        }
240
241        // Show available files
242        let dir = entry.path();
243        let files = ["settings.json", "settings-full.json", "CLAUDE.md", "installed_skills.json"];
244        let mut available = Vec::new();
245        for file in files {
246            if dir.join(file).exists() {
247                available.push(file);
248            }
249        }
250        if dir.join("hooks").exists() {
251            available.push("hooks/");
252        }
253
254        if !available.is_empty() {
255            println!("    文件: {}", available.join(", ").dimmed());
256        }
257        println!();
258    }
259
260    if !found_any {
261        println!("{}", "  没有找到设备配置".dimmed());
262        println!();
263        println!("运行 {} 推送当前设备配置", format!("{} config push", BINARY_NAME).cyan());
264    }
265
266    Ok(())
267}

269/// Apply configuration from another device
270pub fn handle_config_apply(
271    source_device: &str,
272    with_hooks: bool,
273    settings: &ConfigSyncSettings,
274) -> Result<()> {
275    let sync_state = SyncState::load()?;
276    let source_dir = device_config_dir(&sync_state.sync_repo_path, source_device);
277
278    if !source_dir.exists() {
279        return Err(anyhow::anyhow!(
280            "设备配置不存在: {}\n运行 `{} config list` 查看可用配置",
281            source_device, BINARY_NAME
282        ));
283    }
284
285    let claude = claude_dir()?;
286    let current_platform = Platform::current();
287    let mut applied_files = Vec::new();
288
289    println!(
290        "{}",
291        format!("从 {} 应用配置...", source_device).cyan()
292    );
293
294    // Apply settings.json
295    if settings.sync_settings {
296        let settings_file = if with_hooks {
297            "settings-full.json"
298        } else {
299            "settings.json"
300        };
301
302        let source_settings = source_dir.join(settings_file);
303        if source_settings.exists() {
304            let target_settings = claude.join("settings.json");
305
306            // Backup current settings
307            if target_settings.exists() {
308                let backup = claude.join("settings.json.backup");
309                fs::copy(&target_settings, &backup)?;
310                println!("  {} 已备份到 settings.json.backup", "ℹ".blue());
311            }
312
313            if with_hooks {
314                // Copy full version directly
315                fs::copy(&source_settings, &target_settings)?;
316            } else {
317                // Merge: keep local hooks, use remote settings
318                let source_content = fs::read_to_string(&source_settings)?;
319                let target_content = if target_settings.exists() {
320                    fs::read_to_string(&target_settings)?
321                } else {
322                    "{}".to_string()
323                };
324
325                let source_json: serde_json::Value = serde_json::from_str(&source_content)?;
326                let target_json: serde_json::Value = serde_json::from_str(&target_content)?;
327
328                // Merge: source settings + local hooks
329                let mut merged = source_json.clone();
330                if let (Some(merged_obj), Some(target_obj)) = (merged.as_object_mut(), target_json.as_object()) {
331                    if let Some(hooks) = target_obj.get("hooks") {
332                        merged_obj.insert("hooks".to_string(), hooks.clone());
333                    }
334                }
335
336                let merged_content = serde_json::to_string_pretty(&merged)?;
337                fs::write(&target_settings, merged_content)?;
338            }
339
340            applied_files.push(format!("{} ({})", "settings.json", if with_hooks { "含 hooks" } else { "保留本地 hooks" }));
341        }
342    }
343
344    // Apply CLAUDE.md with platform filtering and merging
345    if settings.sync_claude_md {
346        let source_claude_md = source_dir.join("CLAUDE.md");
347        if source_claude_md.exists() {
348            let source_content = fs::read_to_string(&source_claude_md)?;
349            let target_claude_md = claude.join("CLAUDE.md");
350
351            // Backup
352            if target_claude_md.exists() {
353                let backup = claude.join("CLAUDE.md.backup");
354                fs::copy(&target_claude_md, &backup)?;
355            }
356
357            // Read target content (if exists)
358            let target_content = if target_claude_md.exists() {
359                fs::read_to_string(&target_claude_md)?
360            } else {
361                String::new()
362            };
363
364            // Merge: source common content + target's current platform block
365            let final_content = if has_platform_blocks(&source_content) || has_platform_blocks(&target_content) {
366                let merged = merge_claude_md(&source_content, &target_content, current_platform);
367                println!(
368                    "  {} 已合并 CLAUDE.md（保留本地 {} 平台内容）",
369                    "ℹ".blue(),
370                    current_platform
371                );
372                merged
373            } else {
374                // No platform blocks, just use source
375                source_content
376            };
377
378            fs::write(&target_claude_md, final_content)?;
379            applied_files.push("CLAUDE.md".to_string());
380        }
381    }
382
383    // Apply hooks if requested
384    if with_hooks && settings.sync_hooks {
385        let source_hooks = source_dir.join("hooks");
386        if source_hooks.exists() && source_hooks.is_dir() {
387            let target_hooks = claude.join("hooks");
388            fs::create_dir_all(&target_hooks)?;
389
390            for entry in fs::read_dir(&source_hooks)? {
391                let entry = entry?;
392                let target_file = target_hooks.join(entry.file_name());
393                fs::copy(entry.path(), &target_file)?;
394
395                // Make executable on Unix
396                #[cfg(unix)]
397                {
398                    use std::os::unix::fs::PermissionsExt;
399                    let mut perms = fs::metadata(&target_file)?.permissions();
400                    perms.set_mode(0o755);
401                    fs::set_permissions(&target_file, perms)?;
402                }
403            }
404
405            applied_files.push("hooks/".to_string());
406
407            println!();
408            println!("{}", "⚠️  Hooks 已应用，请检查以下路径是否适用于本设备:".yellow());
409            println!("    - ~/.claude/hooks/ 中的脚本内容");
410            println!("    - settings.json 中的 hooks 命令路径");
411        }
412    }
413
414    // Show skills to install
415    let skills_path = source_dir.join("installed_skills.json");
416    if skills_path.exists() {
417        let content = fs::read_to_string(&skills_path)?;
418        if let Ok(skills_list) = serde_json::from_str::<SkillsList>(&content) {
419            if !skills_list.skills.is_empty() {
420                println!();
421                println!("{}", "Skills 安装命令:".cyan());
422                for (_name, url) in &skills_list.skills {
423                    println!("  claude skill install {}", url);
424                }
425            }
426        }
427    }
428
429    // Show plugins to install
430    let plugins_path = source_dir.join("installed_plugins.json");
431    if plugins_path.exists() {
432        if let Ok(content) = fs::read_to_string(&plugins_path) {
433            if let Ok(json) = serde_json::from_str::<serde_json::Value>(&content) {
434                if let Some(plugins) = json.get("plugins").and_then(|p| p.as_object()) {
435                    if !plugins.is_empty() {
436                        println!();
437                        println!("{}", "Plugins 安装命令:".cyan());
438                        for name in plugins.keys() {
439                            println!("  claude plugin install {}", name);
440                        }
441                    }
442                }
443            }
444        }
445    }
446
447    println!();
448    if !applied_files.is_empty() {
449        println!("{}", "✓ 配置已应用".green());
450        for file in &applied_files {
451            println!("  - {}", file);
452        }
453        println!();
454        println!("{}", "请重启 Claude Code 使配置生效".cyan());
455    } else {
456        println!("{}", "没有应用任何配置".yellow());
457    }
458
459    Ok(())
460}

524/// Generate skills list from skills directory
525fn generate_skills_list(skills_dir: &Path) -> Result<SkillsList> {
526    let mut skills = HashMap::new();
527
528    for entry in fs::read_dir(skills_dir)? {
529        let entry = entry?;
530        if !entry.file_type()?.is_dir() {
531            continue;
532        }
533
534        let skill_name = entry.file_name().to_string_lossy().to_string();
535        let git_dir = entry.path().join(".git");
536
537        if git_dir.exists() {
538            // Try to get git remote origin URL
539            if let Ok(output) = std::process::Command::new("git")
540                .args(["remote", "get-url", "origin"])
541                .current_dir(entry.path())
542                .output()
543            {
544                if output.status.success() {
545                    let url = String::from_utf8_lossy(&output.stdout).trim().to_string();
546                    if !url.is_empty() {
547                        skills.insert(skill_name, url);
548                    }
549                }
550            }
551        }
552    }
553
554    Ok(SkillsList { skills })
555}

557/// Recursively copy a directory
558fn copy_dir_recursive(src: &Path, dst: &Path) -> Result<()> {
559    fs::create_dir_all(dst)?;
560
561    for entry in fs::read_dir(src)? {
562        let entry = entry?;
563        let src_path = entry.path();
564        let dst_path = dst.join(entry.file_name());
565
566        if src_path.is_dir() {
567            copy_dir_recursive(&src_path, &dst_path)?;
568        } else {
569            fs::copy(&src_path, &dst_path)?;
570        }
571    }
572
573    Ok(())
574}

581/// Find the most recently synced device config (excluding current device),
582/// returning both device name and its sync timestamp.
583fn find_latest_device_config_with_time(
584    sync_repo: &Path,
585    current_device: &str,
586) -> Option<(String, chrono::DateTime<chrono::Utc>)> {
587    let configs = configs_dir(sync_repo);
588    if !configs.exists() {
589        return None;
590    }
591
592    let mut latest: Option<(String, chrono::DateTime<chrono::Utc>)> = None;
593
594    for entry in fs::read_dir(&configs).ok()?.filter_map(|e| e.ok()) {
595        if !entry.path().is_dir() {
596            continue;
597        }
598
599        let device_name = match entry.file_name().into_string() {
600            Ok(name) => name,
601            Err(_) => continue,
602        };
603
604        // Skip current device
605        if device_name == current_device {
606            continue;
607        }
608
609        // Read .sync-info.json
610        let info_path = entry.path().join(".sync-info.json");
611        if let Ok(content) = fs::read_to_string(&info_path) {
612            if let Ok(info) = serde_json::from_str::<DeviceSyncInfo>(&content) {
613                if let Ok(sync_time) = chrono::DateTime::parse_from_rfc3339(&info.last_sync) {
614                    let sync_time = sync_time.with_timezone(&chrono::Utc);
615                    if latest.is_none() || sync_time > latest.as_ref().unwrap().1 {
616                        latest = Some((device_name, sync_time));
617                    }
618                }
619            }
620        }
621    }
622
623    latest
624}

626/// Get the sync timestamp of a specific device from its .sync-info.json.
627fn get_device_sync_time(
628    sync_repo: &Path,
629    device: &str,
630) -> Option<chrono::DateTime<chrono::Utc>> {
631    let info_path = device_config_dir(sync_repo, device).join(".sync-info.json");
632    let content = fs::read_to_string(&info_path).ok()?;
633    let info: DeviceSyncInfo = serde_json::from_str(&content).ok()?;
634    chrono::DateTime::parse_from_rfc3339(&info.last_sync)
635        .ok()
636        .map(|t| t.with_timezone(&chrono::Utc))
637}

639/// Auto-apply CLAUDE.md from the most recently updated device
640/// Only applies CLAUDE.md, not other config files (settings, hooks, skills)
641/// Only applies if the other device's config is newer than the current device's config
642pub fn auto_apply_claude_md(settings: &ConfigSyncSettings) -> Result<()> {
643    if !settings.enabled || !settings.auto_apply_claude_md {
644        log::debug!("Auto-apply CLAUDE.md is disabled");
645        return Ok(());
646    }
647
648    let sync_state = SyncState::load()?;
649    let current_device = settings.get_device_name();
650
651    // Find most recently updated device (with timestamp)
652    let (latest_device, latest_time) =
653        match find_latest_device_config_with_time(&sync_state.sync_repo_path, &current_device) {
654            Some(d) => d,
655            None => {
656                log::debug!("No other device configs found for auto-apply");
657                return Ok(());
658            }
659        };
660
661    // Only apply if the other device's config is newer than current device's
662    if let Some(current_time) = get_device_sync_time(&sync_state.sync_repo_path, &current_device) {
663        if latest_time <= current_time {
664            log::debug!(
665                "Current device config ({}) is newer than {} ({}), skipping auto-apply",
666                current_time,
667                latest_device,
668                latest_time
669            );
670            return Ok(());
671        }
672    }
673
674    let source_dir = device_config_dir(&sync_state.sync_repo_path, &latest_device);
675    let source_claude_md = source_dir.join("CLAUDE.md");
676
677    if !source_claude_md.exists() {
678        log::debug!("No CLAUDE.md found in device config: {}", latest_device);
679        return Ok(());
680    }
681
682    let claude = claude_dir()?;
683    let target_claude_md = claude.join("CLAUDE.md");
684    let source_content = fs::read_to_string(&source_claude_md)?;
685
686    // Read target content
687    let target_content = if target_claude_md.exists() {
688        fs::read_to_string(&target_claude_md)?
689    } else {
690        String::new()
691    };
692
693    // Only apply if there are platform blocks to merge
694    if has_platform_blocks(&source_content) || has_platform_blocks(&target_content) {
695        let current_platform = Platform::current();
696        let merged = merge_claude_md(&source_content, &target_content, current_platform);
697
698        // Only write if content changed
699        if merged != target_content {
700            fs::write(&target_claude_md, &merged)?;
701            log::info!("Auto-applied CLAUDE.md from device: {}", latest_device);
702        }
703    } else {
704        // No platform blocks - check if source is different and update
705        if source_content != target_content {
706            fs::write(&target_claude_md, &source_content)?;
707            log::info!("Auto-applied CLAUDE.md from device: {}", latest_device);
708        }
709    }
710
711    Ok(())
712}
