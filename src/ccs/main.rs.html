493fn main() -> Result<()> {
494    // Initialize logging (rotate log if needed, then set up logger)
495    logger::rotate_log_if_needed().ok(); // Ignore errors during log rotation
496    logger::init_logger().ok(); // Ignore errors during logger init
497
498    log::debug!("ccs started");
499
500    // Background update check (non-blocking)
501    // Only check if not running update command itself
502    let update_check_handle = std::thread::spawn(|| {
503        if let Some(new_version) = check_for_update_silent() {
504            // Return the version to print after command parsing
505            Some(new_version)
506        } else {
507            None
508        }
509    });
510
511    let cli = Cli::parse();
512
513    // Check if this is the update command (skip notification for update command)
514    let is_update_command = matches!(cli.command, Some(Commands::Update { .. }));
515
516    // Print update notification if available (and not running update command)
517    if !is_update_command {
518        if let Ok(Some(new_version)) = update_check_handle.join() {
519            print_update_notification(&new_version);
520        }
521    }
522
523    // Check if initialization is needed (before processing any command)
524    let needs_onboarding = !is_initialized()?;
525
526    // Determine the actual command to run
527    let command = if let Some(cmd) = cli.command {
528        cmd
529    } else {
530        // No command provided
531        if needs_onboarding {
532            // Will trigger onboarding below, then default to sync
533            Commands::Sync {
534                message: None,
535                branch: None,
536                exclude_attachments: false,
537                interactive: false,
538                verbose: false,
539                quiet: false,
540            }
541        } else {
542            // Already initialized, default to sync
543            Commands::Sync {
544                message: None,
545                branch: None,
546                exclude_attachments: false,
547                interactive: false,
548                verbose: false,
549                quiet: false,
550            }
551        }
552    };
553
554    // Check if this is a command that should skip auto-onboarding
555    let is_init_command = matches!(command, Commands::Init { .. });
556    let is_config_command = matches!(command, Commands::Config { .. });
557    let is_session_command = matches!(command, Commands::Session { .. });
558
559    // Run onboarding if needed (skip for Init, Config, and Session commands)
560    // Session is a local-only feature that doesn't require sync repo configuration
561    if needs_onboarding && !is_init_command && !is_config_command && !is_session_command {
562        log::info!("Running onboarding flow - first time setup detected");
563
564        // Try non-interactive init first (from config file)
565        let mut initialized = try_init_from_config().unwrap_or(false);
566
567        // If no config file, try to recover existing repo
568        if !initialized {
569            initialized = handlers::onboarding::try_recover_existing_repo().unwrap_or(false);
570        }
571
572        if !initialized {
573            // Fall back to interactive onboarding
574            run_onboarding_flow()?;
575        }
576
577        log::info!("Onboarding completed successfully");
578    }
579
580    match command {
581        Commands::Init { local, remote, clone, config } => {
582            // If config file is provided, use non-interactive init
583            if config.is_some() {
584                run_init_from_config(config)?;
585            } else if clone {
586                // Clone mode: requires remote URL
587                let remote_url = remote.as_deref().ok_or_else(|| {
588                    anyhow::anyhow!("--clone requires --remote <URL> to be specified")
589                })?;
590
591                // Determine clone destination
592                let clone_path = if let Some(ref path) = local {
593                    path.clone()
594                } else {
595                    config::ConfigManager::default_repo_dir()?
596                };
597
598                println!(
599                    "{}",
600                    format!("Cloning from {} to {}...", remote_url, clone_path.display()).cyan()
601                );
602
603                scm::clone(remote_url, &clone_path)?;
604                sync::init_from_onboarding(&clone_path, Some(remote_url), true)?;
605
606                // Save default filter configuration if it doesn't exist
607                let filter_config_path = config::ConfigManager::filter_config_path()?;
608                if !filter_config_path.exists() {
609                    filter::FilterConfig::default().save()?;
610                }
611
612                println!(
613                    "{}",
614                    "Clone and initialization complete!".green().bold()
615                );
616            } else if let Some(local_path) = local {
617                // Use CLI args for init (local path)
618                sync::init_sync_repo(&local_path, remote.as_deref())?;
619            } else if let Some(remote_url) = remote {
620                // Just --remote provided: clone to default location
621                let default_path = config::ConfigManager::default_repo_dir()?;
622
623                println!(
624                    "{}",
625                    format!("Cloning from {} to {}...", remote_url, default_path.display()).cyan()
626                );
627
628                scm::clone(&remote_url, &default_path)?;
629                sync::init_from_onboarding(&default_path, Some(&remote_url), true)?;
630
631                // Save default filter configuration if it doesn't exist
632                let filter_config_path = config::ConfigManager::filter_config_path()?;
633                if !filter_config_path.exists() {
634                    filter::FilterConfig::default().save()?;
635                }
636
637                println!(
638                    "{}",
639                    "Clone and initialization complete!".green().bold()
640                );
641            } else {
642                // No args provided, try config file first, then fall back to interactive onboarding
643                if !try_init_from_config()? {
644                    // No config file found, run interactive onboarding
645                    run_onboarding_flow()?;
646                }
647            }
648        }
649        Commands::Push {
650            message,
651            push_remote,
652            branch,
653            exclude_attachments,
654            no_config,
655            interactive,
656            verbose,
657            quiet,
658        } => {
659            // Determine verbosity level
660            let verbosity = if verbose {
661                VerbosityLevel::Verbose
662            } else if quiet {
663                VerbosityLevel::Quiet
664            } else {
665                VerbosityLevel::Normal
666            };
667
668            sync::push_history(
669                message.as_deref(),
670                push_remote,
671                branch.as_deref(),
672                exclude_attachments,
673                !no_config, // sync_config = !no_config
674                interactive,
675                verbosity,
676            )?;
677        }
678        Commands::Pull {
679            fetch_remote,
680            branch,
681            interactive,
682            verbose,
683            quiet,
684        } => {
685            // Determine verbosity level
686            let verbosity = if verbose {
687                VerbosityLevel::Verbose
688            } else if quiet {
689                VerbosityLevel::Quiet
690            } else {
691                VerbosityLevel::Normal
692            };
693
694            sync::pull_history(fetch_remote, branch.as_deref(), interactive, verbosity)?;
695        }
696        Commands::Sync {
697            message,
698            branch,
699            exclude_attachments,
700            interactive,
701            verbose,
702            quiet,
703        } => {
704            // Determine verbosity level
705            let verbosity = if verbose {
706                VerbosityLevel::Verbose
707            } else if quiet {
708                VerbosityLevel::Quiet
709            } else {
710                VerbosityLevel::Normal
711            };
712
713            sync::sync_bidirectional(
714                message.as_deref(),
715                branch.as_deref(),
716                exclude_attachments,
717                interactive,
718                verbosity,
719            )?;
720        }
721        Commands::Status {
722            show_conflicts,
723            show_files,
724        } => {
725            sync::show_status(show_conflicts, show_files)?;
726        }
727        Commands::Config {
728            exclude_older_than,
729            include_projects,
730            exclude_projects,
731            exclude_attachments,
732            enable_lfs,
733            lfs_patterns,
734            scm_backend,
735            sync_subdirectory,
736            use_project_name_only,
737            show,
738            interactive,
739            wizard,
740        } => {
741            // Check if ANY flag was provided
742            let has_any_flag = exclude_older_than.is_some()
743                || include_projects.is_some()
744                || exclude_projects.is_some()
745                || exclude_attachments.is_some()
746                || enable_lfs.is_some()
747                || lfs_patterns.is_some()
748                || scm_backend.is_some()
749                || sync_subdirectory.is_some()
750                || show
751                || interactive
752                || wizard;
753
754            if !has_any_flag {
755                // No args provided - show repo selector menu
756                handle_repo_selector()?;
757            } else if interactive {
758                handle_config_interactive()?;
759            } else if wizard {
760                handle_config_wizard()?;
761            } else if show {
762                filter::show_config()?;
763            } else {
764                filter::update_config(
765                    exclude_older_than,
766                    include_projects,
767                    exclude_projects,
768                    exclude_attachments,
769                    enable_lfs,
770                    lfs_patterns,
771                    scm_backend,
772                    sync_subdirectory,
773                    use_project_name_only,
774                )?;
775            }
776        }
777        Commands::Report { format, output } => {
778            report::generate_report(&format, output.as_deref())?;
779        }
780        Commands::Remote { action } => match action {
781            RemoteAction::Show => {
782                sync::show_remote()?;
783            }
784            RemoteAction::Set { name, url } => {
785                sync::set_remote(&name, &url)?;
786            }
787            RemoteAction::Remove { name } => {
788                sync::remove_remote(&name)?;
789            }
790        },
791        Commands::Undo { operation, verbose, quiet } => {
792            // Determine verbosity level
793            let verbosity = if verbose {
794                VerbosityLevel::Verbose
795            } else if quiet {
796                VerbosityLevel::Quiet
797            } else {
798                VerbosityLevel::Normal
799            };
800
801            match operation {
802                UndoOperation::Pull { preview } => {
803                    handle_undo_pull(preview, verbosity)?;
804                }
805                UndoOperation::Push { preview } => {
806                    handle_undo_push(preview, verbosity)?;
807                }
808            }
809        },
810        Commands::History { action } => match action {
811            HistoryAction::List { limit } => {
812                handle_history_list(limit)?;
813            }
814            HistoryAction::Last { operation_type } => {
815                handle_history_last(operation_type.as_deref())?;
816            }
817            HistoryAction::Review { limit } => {
818                handle_history_review(limit)?;
819            }
820            HistoryAction::Clear => {
821                handle_history_clear()?;
822            }
823        },
824        Commands::Setup { skip_sync } => {
825            handle_setup(skip_sync)?;
826        }
827        Commands::Update { check_only } => {
828            handle_update(check_only)?;
829        }
830        Commands::CleanupSnapshots {
831            dry_run,
832            max_count,
833            max_age_days,
834            interactive,
835            verbose,
836            quiet,
837        } => {
838            // Determine verbosity level
839            let verbosity = if verbose {
840                VerbosityLevel::Verbose
841            } else if quiet {
842                VerbosityLevel::Quiet
843            } else {
844                VerbosityLevel::Normal
845            };
846
847            handle_cleanup_snapshots(dry_run, max_count, max_age_days, interactive, verbosity)?;
848        }
849        Commands::Hooks { action } => match action {
850            HooksAction::Install => {
851                handle_hooks_install()?;
852            }
853            HooksAction::Uninstall => {
854                handle_hooks_uninstall()?;
855            }
856            HooksAction::Show => {
857                handle_hooks_show()?;
858            }
859        },
860        Commands::Wrapper { action } => match action {
861            WrapperAction::Install { force } => {
862                handle_wrapper_install(force)?;
863            }
864            WrapperAction::Uninstall => {
865                handle_wrapper_uninstall()?;
866            }
867            WrapperAction::Show => {
868                handle_wrapper_show()?;
869            }
870        },
871        Commands::Automate { status, uninstall } => {
872            if status {
873                handle_automate_status()?;
874            } else if uninstall {
875                handle_automate_uninstall()?;
876            } else {
877                handle_automate_setup()?;
878            }
879        }
880        Commands::HookNewProjectCheck => {
881            handle_new_project_check()?;
882        }
883        Commands::HookSessionStart => {
884            handle_session_start()?;
885        }
886        Commands::HookStop => {
887            handle_stop()?;
888        }
889        Commands::ConfigSync { action } => {
890            let filter_config = filter::FilterConfig::load()?;
891            match action {
892                ConfigSyncAction::Push => {
893                    handle_config_push(&filter_config.config_sync)?;
894                }
895                ConfigSyncAction::List => {
896                    handle_config_list()?;
897                }
898                ConfigSyncAction::Apply { device, with_hooks } => {
899                    handle_config_apply(&device, with_hooks, &filter_config.config_sync)?;
900                }
901                ConfigSyncAction::Status => {
902                    handle_config_status(&filter_config.config_sync)?;
903                }
904            }
905        }
906        Commands::Session { action, project } => {
907            match action {
908                None => {
909                    // Interactive mode
910                    handle_session_interactive(project.as_deref())?;
911                }
912                Some(SessionAction::List { project: list_project, show_ids }) => {
913                    // Use subcommand project filter if provided, otherwise use global
914                    let filter = list_project.as_deref().or(project.as_deref());
915                    handle_session_list(filter, show_ids)?;
916                }
917                Some(SessionAction::Show { session_id }) => {
918                    handle_session_show(&session_id)?;
919                }
920                Some(SessionAction::Rename { session_id, title }) => {
921                    handle_session_rename(&session_id, &title)?;
922                }
923                Some(SessionAction::Delete { session_id, force }) => {
924                    handle_session_delete(&session_id, force)?;
925                }
926            }
927        }
928    }
929
930    Ok(())
931}
