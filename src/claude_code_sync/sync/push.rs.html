22/// Push local Claude Code history to sync repository
23pub fn push_history(
24    commit_message: Option<&str>,
25    push_remote: bool,
26    branch: Option<&str>,
27    exclude_attachments: bool,
28    sync_config: bool,
29    interactive: bool,
30    verbosity: crate::VerbosityLevel,
31) -> Result<()> {
32    use crate::VerbosityLevel;
33
34    if verbosity != VerbosityLevel::Quiet {
35        println!("{}", "Pushing Claude Code history...".cyan().bold());
36    }
37
38    let state = SyncState::load()?;
39    let repo = scm::open(&state.sync_repo_path)?;
40    let mut filter = FilterConfig::load()?;
41
42    // Override exclude_attachments if specified in command
43    if exclude_attachments {
44        filter.exclude_attachments = true;
45    }
46
47    // Set up LFS if enabled
48    if filter.enable_lfs {
49        if verbosity != VerbosityLevel::Quiet {
50            println!("  {} Git LFS...", "Configuring".cyan());
51        }
52        scm::lfs::setup(&state.sync_repo_path, &filter.lfs_patterns)
53            .context("Failed to set up Git LFS")?;
54    }
55
56    let claude_dir = claude_projects_dir()?;
57
58    // Check directory structure consistency before pushing
59    let projects_dir = state.sync_repo_path.join(&filter.sync_subdirectory);
60    if projects_dir.exists() {
61        let structure_check =
62            check_directory_structure_consistency(&projects_dir, filter.use_project_name_only);
63
64        if !structure_check.is_consistent {
65            if let Some(warning) = &structure_check.warning {
66                println!();
67                println!("{}", "⚠️  目录结构不一致警告".yellow().bold());
68                println!("{}", "─".repeat(50).dimmed());
69                println!("{}", warning.yellow());
70                println!();
71
72                if interactive && interactive_conflict::is_interactive() {
73                    let proceed = Confirm::new("是否继续推送？")
74                        .with_default(false)
75                        .with_help_message("建议先清理目录结构再继续")
76                        .prompt()
77                        .context("取消确认")?;
78
79                    if !proceed {
80                        println!("\n{}", "推送已取消。".yellow());
81                        println!(
82                            "提示：使用 '{}' 可以切换同步模式",
83                            "claude-code-sync config --use-project-name-only <true|false>".cyan()
84                        );
85                        return Ok(());
86                    }
87                } else if verbosity != VerbosityLevel::Quiet {
88                    println!(
89                        "{}",
90                        "使用 --interactive 选项可以在不一致时选择是否继续".dimmed()
91                    );
92                }
93            }
94        }
95    }
96
97    // Get the current branch name for operation record
98    let branch_name = branch
99        .map(|s| s.to_string())
100        .or_else(|| repo.current_branch().ok())
101        .unwrap_or_else(|| "main".to_string());
102
103    // Discover all sessions
104    println!("  {} conversation sessions...", "Discovering".cyan());
105    let sessions = discover_sessions(&claude_dir, &filter)?;
106    println!("  {} {} sessions", "Found".green(), sessions.len());
107
108    // Check for project name collisions when using project-name-only mode
109    if filter.use_project_name_only {
110        let collisions = find_colliding_projects(&claude_dir);
111        if !collisions.is_empty() {
112            println!();
113            println!(
114                "{}",
115                "Warning: Multiple projects map to the same name:".yellow().bold()
116            );
117            for (name, paths) in &collisions {
118                println!("  {} -> {} locations:", name.cyan(), paths.len());
119                for path in paths.iter().take(3) {
120                    let display_path = path.file_name()
121                        .and_then(|n| n.to_str())
122                        .unwrap_or("unknown");
123                    println!("    - {}", display_path);
124                }
125                if paths.len() > 3 {
126                    println!("    ... and {} more", paths.len() - 3);
127                }
128            }
129            println!();
130            println!(
131                "{}",
132                "Sessions from colliding projects will be merged into the same directory.".yellow()
133            );
134            println!();
135        }
136    }
137
138    // ============================================================================
139    // COPY SESSIONS AND TRACK CHANGES
140    // ============================================================================
141    // Note: projects_dir was already defined above for consistency check
142    fs::create_dir_all(&projects_dir)?;
143
144    // Discover existing sessions in sync repo to determine operation type
145    println!("  {} sessions to sync repository...", "Copying".cyan());
146    let existing_sessions = discover_sessions(&projects_dir, &filter)?;
147    let existing_map: HashMap<_, _> = existing_sessions
148        .iter()
149        .map(|s| (s.session_id.clone(), s))
150        .collect();
151
152    // Track pushed conversations for operation record
153    let mut pushed_conversations: Vec<ConversationSummary> = Vec::new();
154    let mut added_count = 0;
155    let mut modified_count = 0;
156    let mut unchanged_count = 0;
157
158    // Track sessions skipped due to missing cwd
159    let mut skipped_no_cwd = 0;
160
161    // Closure to compute the relative path for a session, respecting use_project_name_only
162    let compute_relative_path =
163        |session: &crate::parser::ConversationSession| -> Option<PathBuf> {
164            if filter.use_project_name_only {
165                let full_relative = Path::new(&session.file_path)
166                    .strip_prefix(&claude_dir)
167                    .unwrap_or(Path::new(&session.file_path));
168
169                let filename = full_relative.file_name()?;
170                let project_name = session.project_name()?;
171                Some(PathBuf::from(project_name).join(filename))
172            } else {
173                Some(
174                    Path::new(&session.file_path)
175                        .strip_prefix(&claude_dir)
176                        .unwrap_or(Path::new(&session.file_path))
177                        .to_path_buf(),
178                )
179            }
180        };
181
182    for session in &sessions {
183        let relative_path = match compute_relative_path(session) {
184            Some(path) => path,
185            None => {
186                skipped_no_cwd += 1;
187                log::debug!("Skipping session {} (no cwd)", session.session_id);
188                continue;
189            }
190        };
191
192        let dest_path = projects_dir.join(&relative_path);
193
194        // Determine operation type based on existing state
195        let operation = if let Some(existing) = existing_map.get(&session.session_id) {
196            if existing.content_hash() == session.content_hash() {
197                unchanged_count += 1;
198                SyncOperation::Unchanged
199            } else {
200                modified_count += 1;
201                SyncOperation::Modified
202            }
203        } else {
204            added_count += 1;
205            SyncOperation::Added
206        };
207
208        // Write the session file
209        session.write_to_file(&dest_path)?;
210
211        // Track this session in pushed conversations
212        let relative_path_str = relative_path.to_string_lossy().to_string();
213        match ConversationSummary::new(
214            session.session_id.clone(),
215            relative_path_str.clone(),
216            session.latest_timestamp(),
217            session.message_count(),
218            operation,
219        ) {
220            Ok(summary) => pushed_conversations.push(summary),
221            Err(e) => log::warn!(
222                "Failed to create summary for {}: {}",
223                relative_path_str,
224                e
225            ),
226        }
227    }
228
229    // ============================================================================
230    // SHOW SUMMARY AND INTERACTIVE CONFIRMATION
231    // ============================================================================
232    if verbosity != VerbosityLevel::Quiet {
233        println!();
234        println!("{}", "Push Summary:".bold().cyan());
235        println!("  {} Added: {}", "•".green(), added_count);
236        println!("  {} Modified: {}", "•".yellow(), modified_count);
237        println!("  {} Unchanged: {}", "•".dimmed(), unchanged_count);
238        let total_with_cwd = sessions.len().saturating_sub(skipped_no_cwd);
239        println!("  {} Skipped (no cwd): {}", "•".dimmed(), skipped_no_cwd);
240        println!(
241            "  {} Sessions (with project context): {}",
242            "•".cyan(),
243            total_with_cwd
244        );
245        println!();
246    }
247
248    // Show detailed file list in verbose mode
249    if verbosity == VerbosityLevel::Verbose {
250        println!("{}", "Files to be pushed:".bold());
251        for (idx, session) in sessions.iter().enumerate().take(20) {
252            let Some(relative_path) = compute_relative_path(session) else {
253                continue;
254            };
255
256            let status = if let Some(existing) = existing_map.get(&session.session_id) {
257                if existing.content_hash() == session.content_hash() {
258                    "unchanged".dimmed()
259                } else {
260                    "modified".yellow()
261                }
262            } else {
263                "new".green()
264            };
265
266            println!("  {}. {} [{}]", idx + 1, relative_path.display(), status);
267        }
268        if sessions.len() > 20 {
269            println!("  ... and {} more", sessions.len() - 20);
270        }
271        println!();
272    }
273
274    // Interactive confirmation
275    if interactive && interactive_conflict::is_interactive() {
276        let confirm = Confirm::new("Do you want to proceed with pushing these changes?")
277            .with_default(true)
278            .with_help_message("This will commit and push to the sync repository")
279            .prompt()
280            .context("Failed to get confirmation")?;
281
282        if !confirm {
283            println!("\n{}", "Push cancelled.".yellow());
284            return Ok(());
285        }
286    }
287
288    // ============================================================================
289    // SYNC DEVICE CONFIGURATION (if enabled)
290    // ============================================================================
291    if sync_config && filter.config_sync.enabled && filter.config_sync.push_with_config {
292        if verbosity != VerbosityLevel::Quiet {
293            println!();
294            println!("  {} device configuration...", "Syncing".cyan());
295        }
296
297        // Use config_sync handler to push configuration files (no commit)
298        match crate::handlers::config_sync::push_config_files(&filter.config_sync) {
299            Ok(synced_files) => {
300                if !synced_files.is_empty() {
301                    if verbosity != VerbosityLevel::Quiet {
302                        println!("  {} Device configuration synced:", "✓".green());
303                        for file in &synced_files {
304                            println!("    - {}", file.dimmed());
305                        }
306                    }
307                } else if verbosity == VerbosityLevel::Verbose {
308                    println!("  {} No configuration files to sync", "ℹ".dimmed());
309                }
310            }
311            Err(e) => {
312                log::warn!("Failed to sync device configuration: {}", e);
313                if verbosity != VerbosityLevel::Quiet {
314                    println!(
315                        "  {} Failed to sync device configuration: {}",
316                        "⚠".yellow(),
317                        e
318                    );
319                }
320            }
321        }
322    }
323
324    // ============================================================================
325    // COMMIT AND PUSH CHANGES
326    // ============================================================================
327    repo.stage_all()?;
328
329    let has_changes = repo.has_changes()?;
330    if has_changes {
331        // Get the current commit hash before making any changes
332        // This allows us to undo the push later by resetting to this commit
333        // Note: We don't create file snapshots for push - git already has history!
334        // Undo push simply does `git reset` to this commit.
335        // On a brand new repo with no commits, this will be None (no undo available for first push)
336        let commit_before_push = repo.current_commit_hash().ok();
337
338        if let Some(ref hash) = commit_before_push {
339            if verbosity != VerbosityLevel::Quiet {
340                println!(
341                    "  {} Recorded commit {} for undo",
342                    "✓".green(),
343                    &hash[..8]
344                );
345            }
346        } else if verbosity != VerbosityLevel::Quiet {
347            println!(
348                "  {} First push - no previous commit to undo to",
349                "ℹ".cyan()
350            );
351        }
352
353        let default_message = format!(
354            "Sync {} sessions at {}",
355            sessions.len(),
356            chrono::Utc::now().format("%Y-%m-%d %H:%M:%S UTC")
357        );
358        let message = commit_message.unwrap_or(&default_message);
359
360        println!("  {} changes...", "Committing".cyan());
361        repo.commit(message)?;
362        println!("  {} Committed: {}", "✓".green(), message);
363
364        // Push to remote if configured
365        if push_remote && state.has_remote {
366            println!("  {} to remote...", "Pushing".cyan());
367
368            match repo.push("origin", &branch_name) {
369                Ok(_) => println!("  {} Pushed to origin/{}", "✓".green(), branch_name),
370                Err(e) => log::warn!("Failed to push: {}", e),
371            }
372        }
373
374        // ============================================================================
375        // CREATE AND SAVE OPERATION RECORD
376        // ============================================================================
377        let mut operation_record = OperationRecord::new(
378            OperationType::Push,
379            Some(branch_name.clone()),
380            pushed_conversations.clone(),
381        );
382
383        // Store commit hash for undo (no file snapshot needed - git has history)
384        // On first push (no prior commits), this will be None
385        operation_record.commit_hash = commit_before_push;
386
387        // Load operation history and add this operation
388        let mut history = match OperationHistory::load() {
389            Ok(h) => h,
390            Err(e) => {
391                log::warn!("Failed to load operation history: {}", e);
392                log::info!("Creating new history...");
393                OperationHistory::default()
394            }
395        };
396
397        if let Err(e) = history.add_operation(operation_record) {
398            log::warn!("Failed to save operation to history: {}", e);
399            log::info!("Push completed successfully, but history was not updated.");
400        }
401    } else {
402        println!("  {} No changes to commit", "Note:".yellow());
403    }
404
405    // ============================================================================
406    // DISPLAY SUMMARY TO USER
407    // ============================================================================
408    println!("\n{}", "=== Push Summary ===".bold().cyan());
409
410    // Show operation statistics
411    let stats_msg = format!(
412        "  {} Added    {} Modified    {} Unchanged",
413        format!("{added_count}").green(),
414        format!("{modified_count}").cyan(),
415        format!("{unchanged_count}").dimmed(),
416    );
417    println!("{stats_msg}");
418    println!();
419
420    // Group conversations by project (top-level directory)
421    let mut by_project: HashMap<String, Vec<&ConversationSummary>> = HashMap::new();
422    for conv in &pushed_conversations {
423        // Skip unchanged conversations in detailed output
424        if conv.operation == SyncOperation::Unchanged {
425            continue;
426        }
427
428        let project = conv
429            .project_path
430            .split('/')
431            .next()
432            .unwrap_or("unknown")
433            .to_string();
434        by_project.entry(project).or_default().push(conv);
435    }
436
437    // Display conversations grouped by project
438    if !by_project.is_empty() {
439        println!("{}", "Pushed Conversations:".bold());
440
441        let mut projects: Vec<_> = by_project.keys().collect();
442        projects.sort();
443
444        for project in projects {
445            let conversations = &by_project[project];
446            println!("\n  {} {}/", "Project:".bold(), project.cyan());
447
448            for conv in conversations.iter().take(MAX_CONVERSATIONS_TO_DISPLAY) {
449                let operation_str = match conv.operation {
450                    SyncOperation::Added => "ADD".green(),
451                    SyncOperation::Modified => "MOD".cyan(),
452                    SyncOperation::Conflict => "CONFLICT".yellow(),
453                    SyncOperation::Unchanged => "---".dimmed(),
454                };
455
456                let timestamp_str = conv
457                    .timestamp
458                    .as_ref()
459                    .and_then(|t| {
460                        // Extract just the date portion for compact display
461                        t.split('T').next()
462                    })
463                    .unwrap_or("unknown");
464
465                println!(
466                    "    {} {} ({}msg, {})",
467                    operation_str,
468                    conv.project_path,
469                    conv.message_count,
470                    timestamp_str.dimmed()
471                );
472            }
473
474            if conversations.len() > MAX_CONVERSATIONS_TO_DISPLAY {
475                println!(
476                    "    {} ... and {} more conversations",
477                    "...".dimmed(),
478                    conversations.len() - MAX_CONVERSATIONS_TO_DISPLAY
479                );
480            }
481        }
482    }
483
484    if verbosity == VerbosityLevel::Quiet {
485        println!("Push complete");
486    } else {
487        println!("\n{}", "Push complete!".green().bold());
488    }
489
490    // Clean up old snapshots automatically
491    if let Err(e) = crate::undo::cleanup_old_snapshots(None, false) {
492        log::warn!("Failed to cleanup old snapshots: {}", e);
493    }
494
495    Ok(())
496}
