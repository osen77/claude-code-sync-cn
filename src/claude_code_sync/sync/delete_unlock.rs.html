1//! Time-boxed "delete unlock" window.
2//!
3//! When active, `ccs push` treats locally-missing sessions as intentional
4//! deletions and prunes them from the sync repo (same as `--prune`, but with
5//! NO tombstone). The window expires passively — there is no background
6//! process; every consumer re-checks the stored expiry timestamp.
7
8use crate::config::ConfigManager;
9use anyhow::{Context, Result};
10use serde::{Deserialize, Serialize};
11use std::path::PathBuf;
12use std::time::{SystemTime, UNIX_EPOCH};
13
14#[derive(Debug, Serialize, Deserialize)]
15struct UnlockState {
16    /// Absolute expiry in unix seconds (timezone-independent).
17    expires_at: u64,
18}
19
20fn now_secs() -> u64 {
21    SystemTime::now()
22        .duration_since(UNIX_EPOCH)
23        .map(|d| d.as_secs())
24        // Fail-closed: on clock error, return MAX so `remaining_at` treats the
25        // window as expired rather than reporting it active with a huge
26        // remaining (宁可放行失败，绝不误删).
27        .unwrap_or(u64::MAX)
28}
29
30/// Remaining seconds until expiry, or `None` if already expired.
31/// Pure function — no IO — so expiry logic is unit-testable in isolation.
32fn remaining_at(expires_at: u64, now: u64) -> Option<u64> {
33    if now < expires_at {
34        Some(expires_at - now)
35    } else {
36        None
37    }
38}
39
40fn state_path() -> Result<PathBuf> {
41    ConfigManager::delete_unlock_path()
42}
43
44/// Compute the expiry timestamp for a `minutes`-long window starting at `now`.
45/// Fail-closed on the clock-error sentinel (`u64::MAX` from `now_secs`): refuse
46/// to open the window instead of writing a permanent deadline that would leave
47/// protection off forever (对齐读路径的 fail-closed，绝不因时钟异常永久放行).
48/// Pure function — no IO — so the guard is unit-testable without a real clock.
49fn compute_expiry(now: u64, minutes: u64) -> Result<u64> {
50    if now == u64::MAX {
51        anyhow::bail!("系统时钟异常，无法开启删除放行窗口（请检查系统时间）");
52    }
53    Ok(now.saturating_add(minutes.saturating_mul(60)))
54}
55
56/// Open (or extend) the window for `minutes`. Overwrites any existing state,
57/// so calling again simply renews the deadline. Returns the expiry unix ts.
58pub fn unlock(minutes: u64) -> Result<u64> {
59    let expires_at = compute_expiry(now_secs(), minutes)?;
60    ConfigManager::ensure_config_dir()?;
61    let path = state_path()?;
62    let json = serde_json::to_string(&UnlockState { expires_at })?;
63    std::fs::write(&path, json)
64        .with_context(|| format!("Failed to write delete-unlock state: {}", path.display()))?;
65    Ok(expires_at)
66}
67
68/// Close the window. Idempotent: a missing file is treated as success.
69pub fn disable() -> Result<()> {
70    let path = state_path()?;
71    if path.exists() {
72        std::fs::remove_file(&path)
73            .with_context(|| format!("Failed to remove delete-unlock state: {}", path.display()))?;
74    }
75    Ok(())
76}
77
78/// Remaining seconds if the window is active, else `None` (expired/absent).
79pub fn status() -> Result<Option<u64>> {
80    let path = state_path()?;
81    if !path.exists() {
82        return Ok(None);
83    }
84    let content = std::fs::read_to_string(&path)
85        .with_context(|| format!("Failed to read delete-unlock state: {}", path.display()))?;
86    let state: UnlockState = serde_json::from_str(&content)
87        .with_context(|| format!("Failed to parse delete-unlock state: {}", path.display()))?;
88    Ok(remaining_at(state.expires_at, now_secs()))
89}
90
91/// Fail-safe active check for push consumption. ANY error (missing/corrupt/
92/// unreadable state) resolves to `false` so push falls back to protection.
93#[allow(dead_code)]
94pub fn is_active() -> bool {
95    matches!(status(), Ok(Some(_)))
96}

98#[cfg(test)]
99mod tests {
100    use super::*;
101    use crate::config::CONFIG_DIR_ENV;
102    use serial_test::serial;
103    use std::env;
104    use tempfile::TempDir;
105
106    fn with_temp_config(f: impl FnOnce() + std::panic::UnwindSafe) {
107        let saved = env::var(CONFIG_DIR_ENV).ok();
108        let tmp = TempDir::new().unwrap();
109        env::set_var(CONFIG_DIR_ENV, tmp.path());
110        let result = std::panic::catch_unwind(f);
111        match saved {
112            Some(v) => env::set_var(CONFIG_DIR_ENV, v),
113            None => env::remove_var(CONFIG_DIR_ENV),
114        }
115        if let Err(e) = result {
116            std::panic::resume_unwind(e);
117        }
118    }

120    #[test]
121    fn test_remaining_at_active() {
122        assert_eq!(remaining_at(100, 40), Some(60));
123    }
124
125    #[test]
126    fn test_remaining_at_expired() {
127        assert_eq!(remaining_at(100, 100), None);
128        assert_eq!(remaining_at(100, 150), None);
129    }
130
131    #[test]
132    fn test_compute_expiry_normal() {
133        // 15 minutes = 900s past `now`.
134        assert_eq!(compute_expiry(1000, 15).unwrap(), 1900);
135    }
136
137    #[test]
138    fn test_compute_expiry_saturates_no_panic() {
139        // Huge --minutes must saturate rather than overflow-panic.
140        assert_eq!(compute_expiry(1000, u64::MAX).unwrap(), u64::MAX);
141    }
142
143    #[test]
144    fn test_compute_expiry_clock_error_fails_closed() {
145        // now_secs() sentinel for a clock error must refuse to open the window,
146        // never write a permanent (u64::MAX) deadline.
147        assert!(compute_expiry(u64::MAX, 15).is_err());
148    }
149
150    #[test]
151    #[serial]
152    fn test_unlock_then_status_roundtrip() {
153        with_temp_config(|| {
154            unlock(15).unwrap();
155            let remaining = status().unwrap().expect("window should be active");
156            // 15 minutes = 900s; allow a little slack for test execution.
157            assert!(remaining > 890 && remaining <= 900, "remaining={remaining}");
158            assert!(is_active());
159        });
160    }
161
162    #[test]
163    #[serial]
164    fn test_disable_clears_window() {
165        with_temp_config(|| {
166            unlock(15).unwrap();
167            disable().unwrap();
168            assert_eq!(status().unwrap(), None);
169            assert!(!is_active());
170        });
171    }

173    #[test]
174    #[serial]
175    fn test_absent_file_is_inactive() {
176        with_temp_config(|| {

178            assert!(!is_active());
179        });
180    }

182    #[test]
183    #[serial]
184    fn test_corrupt_file_is_failsafe_inactive() {
185        with_temp_config(|| {
186            ConfigManager::ensure_config_dir().unwrap();
187            std::fs::write(ConfigManager::delete_unlock_path().unwrap(), "not json {{").unwrap();
188            // status() surfaces the parse error, but is_active() must be fail-safe.
