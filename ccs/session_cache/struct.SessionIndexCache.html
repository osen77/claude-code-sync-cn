impl SessionIndexCache

Source

fn empty() -> Self

Create an empty cache with the current version.

Source

pub fn load(config_dir: &Path) -> Self

Load the cache from {config_dir}/session_index.json.

Returns an empty cache on any error (missing file, parse failure, version mismatch). Never panics.

Source

pub fn save(&self, config_dir: &Path)

Save the cache to {config_dir}/session_index.json.

Creates config_dir if it does not exist. Logs warnings on error but does not propagate them — the cache is advisory.

Source

pub fn lookup( &self, key: &str, file_path: &Path, file_size: u64, mtime_secs: i64, ) -> Option<SessionSummary>

Return a cached SessionSummary for the given file only if both file_size and mtime_secs match the stored entry exactly.

key must be the same string used for insert (typically file_path.to_string_lossy()).

Source

pub fn insert( &mut self, key: String, file_size: u64, mtime_secs: i64, summary: &SessionSummary, )

Insert or update a cache entry.

key should be file_path.to_string_lossy() — the same value used for lookup and retain_existing.

Source

pub fn retain_existing(&mut self, seen_paths: &HashSet<String>)

Remove all entries whose keys are not present in seen_paths.

Call this after a full scan to evict stale entries for deleted files.
